🎓 EXEMPLO RÁPIDO (copy-paste):

  import asyncio
  from vexus_crm.agents.proposal_generator import ProposalGeneratorAgent
  from vexus_crm.agents.proposal_models import ProposalRequest, LineItem

  async def criar_proposta():
      agent = ProposalGeneratorAgent()
      
      req = ProposalRequest(
          contact_id="cli_001",
          contact_name="João",
          contact_email="joao@test.com",
          contact_phone="+5511999999",
          line_items=[LineItem(
              product_id="prod_001",
              product_name="Consultoria",
              quantity=50,
              unit_price=5000
          )]
      )
      
      proposal = await agent.generate(req)
      await agent.send_proposal(proposal.id, "whatsapp")
      print(f"✅ Proposta enviada: R${proposal.total:,.2f}")

  asyncio.run(criar_proposta())
